async function doLogin(){
  const key=document.getElementById('loginKey').value.trim();

    const r=await fetch(API+'/admin/api/verify',{headers:{'Authorization':'Bearer '+key}});

(async function autoLogin(){

    // First check if proxy requires authentication
    const testR=await fetch(API+'/admin/api/verify');
    if(testR.ok){
      const d=await testR.json().catch(()=>({}));

    authToken=saved;
    document.getElementById('loginWrap').style.display='none';
    document.getElementById('app').style.display='flex';
    initDashboard();
  }catch{}
})();

  document.getElementById('oauthFlowStatus').style.display='block';
  document.getElementById('oauthDot').className='oauth-dot off';

  document.getElementById('oauthUrlBox').style.display='none';
  document.getElementById('callbackResult').style.display='none';
  try{
    const r=await fetch(API+'/admin/api/oauth/init',{method:'POST',headers:authHeaders(),body:JSON.stringify({provider,plan})});
    const d=await r.json();
    if(!r.ok)throw new Error(d.error||'Init failed');
    currentFlowId=d.flowId;
    sessionStorage.setItem('zcodeOAuthFlowId',d.flowId);
    document.getElementById('oauthDot').className='oauth-dot on';
    document.getElementById('oauthStatusText').textContent='等待授权中...';
    document.getElementById('oauthUrlBox').style.display='block';
    document.getElementById('oauthUrl').value=d.authorizeUrl;
    // Use actual expiry from server if available (expiresAt in ms)
    oauthExpiresAt=d.expiresAt||0;
    startOauthCountdown();
    pollOAuth(d.flowId);
  }catch(e){toast(e.message,'error');document.getElementById('oauthStatusText').textContent='失败：'+e.message}
}
function startOauthCountdown(){
  if(oauthExpiryTimer)clearInterval(oauthExpiryTimer);
  const update=()=>{
    const remainingMs=oauthExpiresAt?oauthExpiresAt-Date.now():0;
    if(remainingMs<=0){
      document.getElementById('oauthCountdown').textContent='已过期';
      clearInterval(oauthExpiryTimer);
      return;
    }
    const s=Math.floor(remainingMs/1000);
    const m=Math.floor(s/60),ss=s%60;
    document.getElementById('oauthCountdown').textContent=`${m}:${String(ss).padStart(2,'0')}`;
  };
  update();
  oauthExpiryTimer=setInterval(update,1000);
}
async function pollOAuth(flowId){
  for(let i=0;i<120;i++){
    await new Promise(r=>setTimeout(r,2000));
    try{
      const r=await fetch(API+'/admin/api/oauth/poll?flowId='+flowId,{headers:authHeaders()});
      const d=await r.json();
      if(d.status==='ready'){
        document.getElementById('oauthDot').className='oauth-dot on';
        document.getElementById('oauthStatusText').innerHTML='<span class="badge badge-success">已授权！</span>';
        document.getElementById('oauthUrlBox').style.display='none';
        if(oauthExpiryTimer)clearInterval(oauthExpiryTimer);
        toast('OAuth 登录成功','success');
        currentFlowId=null;
        sessionStorage.removeItem('zcodeOAuthFlowId');
        loadAccountInfo();loadCredStore();
        return;
      }
      if(d.status==='failed'){
        document.getElementById('oauthDot').className='oauth-dot off';
        document.getElementById('oauthStatusText').textContent='授权失败';
        if(oauthExpiryTimer)clearInterval(oauthExpiryTimer);
        toast('OAuth 失败','error');return;
      }
    }catch(e){console.error('pollOAuth fetch failed:',e)}
  }
  document.getElementById('oauthStatusText').textContent='已超时（仍可在下方粘贴回调 URL）';
  if(oauthExpiryTimer)clearInterval(oauthExpiryTimer);
}

let logEventSource=null;
function connectLogStream(){
  // Backend serves /admin/api/logs/stream as SSE (text/event-stream), not WebSocket.
  // Use EventSource to consume it. SSE doesn't allow custom headers, so the proxy
  // API key is passed as a query parameter for auth.
  if(logEventSource){try{logEventSource.close()}catch{}}
  const sseUrl=API+'/admin/api/logs/stream?token='+encodeURIComponent(authToken);
  try{
    logEventSource=new EventSource(sseUrl);
    logEventSource.onmessage=e=>{
      try{
        const d=JSON.parse(e.data);
        appendLog(d);
      }catch(err){console.error('log SSE parse failed:',err,e.data)}
    };
    logEventSource.onerror=()=>{
      try{logEventSource.close()}catch(err){void err}
      // Reconnect after delay
      setTimeout(()=>{if(authToken)connectLogStream()},3000);
    };
  }catch(err){console.error('connectLogStream failed:',err)}
}
function appendLog(entry){
  logLines.push(entry);
  if(logLines.length>2000)logLines=logLines.slice(-1000);
  renderLogs();
}
function renderLogs(){
  const box=document.getElementById('logBox');
  const level=document.getElementById('logLevel').value;
  const search=document.getElementById('logSearch').value.toLowerCase();
  const filtered=logLines.filter(l=>{
    if(level&&l.level!==level)return false;
    if(search&&!l.message.toLowerCase().includes(search))return false;
    return true;
  });
  box.textContent=filtered.map(l=>`[${l.time}] [${l.level||'info'}] ${l.message}`).join('\n');
  if(document.getElementById('logAutoScroll').checked)box.scrollTop=box.scrollHeight;
}
function clearLogs(){logLines=[];document.getElementById('logBox').textContent=''}
document.getElementById('logLevel').addEventListener('change',renderLogs);
document.getElementById('logSearch').addEventListener('input',renderLogs);

function renderRoutingRules(){
  const tb=document.getElementById('routingRules');
  if(!routingRules.length){tb.innerHTML='<tr><td colspan="5" class="empty-state">暂无自定义路由规则</td></tr>';return}
  tb.innerHTML=routingRules.map((r,i)=>`<tr>
    <td><input value="${escapeHtml(r.pattern)}" onchange="routingRules[${i}].pattern=this.value" placeholder="例如 glm-5*" style="width:100%;padding:4px 8px;background:var(--bg);color:var(--text);border:1px solid var(--border);border-radius:4px"></td>
    <td><select onchange="routingRules[${i}].provider=this.value" style="padding:4px 8px;background:var(--bg);color:var(--text);border:1px solid var(--border);border-radius:4px"><option value="zai"${r.provider==='zai'?' selected':''}>Z.AI</option><option value="bigmodel"${r.provider==='bigmodel'?' selected':''}>Bigmodel</option></select></td>
    <td><input value="${escapeHtml(r.endpoint||'')}" onchange="routingRules[${i}].endpoint=this.value" placeholder="默认" style="width:100%;padding:4px 8px;background:var(--bg);color:var(--text);border:1px solid var(--border);border-radius:4px"></td>
    <td><input value="${escapeHtml(r.note||'')}" onchange="routingRules[${i}].note=this.value" placeholder="可选" style="width:100%;padding:4px 8px;background:var(--bg);color:var(--text);border:1px solid var(--border);border-radius:4px"></td>
    <td><button class="btn btn-danger btn-sm" onclick="routingRules.splice(${i},1);renderRoutingRules()">删除</button></td>
  </tr>`).join('');
}
async function loadRoutingRules(){
  try{
    const r=await fetch(API+'/admin/api/routing-rules',{headers:authHeaders()});
    const d=await r.json();
    routingRules=(d.rules||[]).map(r=>({pattern:r.pattern,provider:r.provider,endpoint:r.endpoint||'',note:r.note||''}));
    renderRoutingRules();
  }catch(e){console.error('loadRoutingRules failed:',e);toast('加载路由规则失败','error')}
}
async function saveRoutingRules(){
  // Validate before submit
  for(let i=0;i<routingRules.length;i++){
    const r=routingRules[i];
    if(!r.pattern||!r.pattern.trim()){toast(`第 ${i+1} 条规则缺少模型匹配规则`,'error');return}
    if(r.provider!=='zai'&&r.provider!=='bigmodel'){toast(`第 ${i+1} 条规则的提供商无效`,'error');return}
  }
  try{
    const r=await fetch(API+'/admin/api/routing-rules',{method:'PUT',headers:authHeaders(),body:JSON.stringify({rules:routingRules})});
    const d=await r.json();
    if(r.ok){
      // Replace local rules with server-normalized version
      routingRules=(d.rules||[]).map(r=>({pattern:r.pattern,provider:r.provider,endpoint:r.endpoint||'',note:r.note||''}));
      renderRoutingRules();
      toast(`已保存 ${routingRules.length} 条路由规则`,'success');
    }else{
      toast(d.error||'保存失败','error');
    }
  }catch(e){console.error('saveRoutingRules failed:',e);toast('保存失败','error')}
}
